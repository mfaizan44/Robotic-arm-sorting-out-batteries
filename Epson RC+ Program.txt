! Sample Epson RC+ Program with TCP/IP Listening

1   ! Start of the program
10  TCP_LISTEN 5000  ! Listen on port 5000
20  LABEL WAIT_FOR_SIGNAL
30  TCP_ACCEPT hSocket  ! Accept incoming connection
40  TCP_RECV hSocket, sData  ! Receive data
50  IF sData == "MOVE" THEN
60      MOV L P_TARGET  ! Move to target position
70  ENDIF
80  TCP_CLOSE hSocket
90  GOTO WAIT_FOR_SIGNAL
